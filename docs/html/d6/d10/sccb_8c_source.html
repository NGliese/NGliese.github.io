61 uint8_t SCCB_Probe(void)
62 {
63  uint8_t slave_addr = 0x0;
64  // scan function
65  for(size_t i = 1; i < 0x80; i++)
66  {
67  i2c_cmd_handle_t cmd = i2c_cmd_link_create();
68  i2c_master_start(cmd);
69  i2c_master_write_byte(cmd, (i << 1) | WRITE_BIT, ACK_CHECK_EN);
70  i2c_master_stop(cmd);
71  esp_err_t ret = i2c_master_cmd_begin(SCCB_I2C_PORT, cmd, 1000 / portTICK_RATE_MS);
72  i2c_cmd_link_delete(cmd);
73  if(ret == ESP_OK)
74  {
75  ESP_LOGW(TAG, "Found I2C Device at 0x%02X", i);
76  }
77  }
78  for(size_t i = 0; i < CAMERA_MODEL_MAX; i++)
79  {
80  slave_addr = camera_sensor[i].sccb_addr;
81  if(slave_addr == 0x00)
82  {
83  ESP_LOGD(TAG, " device 0x%02X ignored...", slave_addr);
84  continue;
85  }
86 
87  ESP_LOGD(TAG, "Looking for device 0x%02X", slave_addr);
88 
89  i2c_cmd_handle_t cmd = i2c_cmd_link_create();
90  i2c_master_start(cmd);
91  i2c_master_write_byte(cmd, (slave_addr << 1) | WRITE_BIT, ACK_CHECK_EN);
92  i2c_master_stop(cmd);
93  esp_err_t ret = i2c_master_cmd_begin(SCCB_I2C_PORT, cmd, 1000 / portTICK_RATE_MS);
94  i2c_cmd_link_delete(cmd);
95  if(ret == ESP_OK)
96  {
97  ESP_LOGD(TAG, "device found : 0x%02X", slave_addr);
98  return slave_addr;
99  }
100  }
101  return 0;
102 }
