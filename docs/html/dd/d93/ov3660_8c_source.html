43 static int read_reg16(uint8_t slv_addr, const uint16_t reg)
44 {
45  int ret = 0, ret2 = 0;
46  ret = read_reg(slv_addr, reg);
47  if(ret >= 0)
48  {
49  ret = (ret & 0xFF) << 8;
50  ret2 = read_reg(slv_addr, reg + 1);
51  if(ret2 < 0)
52  {
53  ret = ret2;
54  }
55  else
56  {
57  ret |= ret2 & 0xFF;
58  }
59  }
60  return ret;
61 }

63 static int write_reg(uint8_t slv_addr, const uint16_t reg, uint8_t value)
64 {
65  int ret = 0;
66 #ifndef REG_DEBUG_ON
67  ret = SCCB_Write16(slv_addr, reg, value);
68 #else
69  int old_value = read_reg(slv_addr, reg);
70  if(old_value < 0)
71  {
72  return old_value;
73  }
74  if((uint8_t)old_value != value)
75  {
76  ESP_LOGI(TAG, "NEW REG 0x%04x: 0x%02x to 0x%02x", reg, (uint8_t)old_value, value);
77  ret = SCCB_Write16(slv_addr, reg, value);
78  }
79  else
80  {
81  ESP_LOGD(TAG, "OLD REG 0x%04x: 0x%02x", reg, (uint8_t)old_value);
82  ret = SCCB_Write16(slv_addr, reg, value); // maybe not?
83  }
84  if(ret < 0)
85  {
86  ESP_LOGE(TAG, "WRITE REG 0x%04x FAILED: %d", reg, ret);
87  }
88 #endif
89  return ret;
90 }

92 static int set_reg_bits(uint8_t slv_addr, uint16_t reg, uint8_t offset, uint8_t mask, uint8_t value)
93 {
94  int ret = 0;
95  uint8_t c_value, new_value;
96  ret = read_reg(slv_addr, reg);
97  if(ret < 0)
98  {
99  return ret;
100  }
101  c_value = ret;
102  new_value = (c_value & ~(mask << offset)) | ((value & mask) << offset);
103  ret = write_reg(slv_addr, reg, new_value);
104  return ret;
105 }

107 static int write_regs(uint8_t slv_addr, const uint16_t (*regs)[2])
108 {
109  int i = 0, ret = 0;
110  while(!ret && regs[i][0] != REGLIST_TAIL)
111  {
112  if(regs[i][0] == REG_DLY)
113  {
114  vTaskDelay(regs[i][1] / portTICK_PERIOD_MS);
115  }
116  else
117  {
118  ret = write_reg(slv_addr, regs[i][0], regs[i][1]);
119  }
120  i++;
121  }
122  return ret;
123 }

146 static int calc_sysclk(int xclk, bool pll_bypass, int pll_multiplier, int pll_sys_div,
147  int pll_pre_div, bool pll_root_2x, int pll_seld5, bool pclk_manual,
148  int pclk_div)
149 {
150  const int pll_pre_div2x_map[] = {2, 3, 4, 6}; // values are multiplied by two to avoid floats
151  const int pll_seld52x_map[] = {2, 2, 4, 5};
152 
153  if(!pll_sys_div)
154  {
155  pll_sys_div = 1;
156  }
157 
158  int pll_pre_div2x = pll_pre_div2x_map[pll_pre_div];
159  int pll_root_div = pll_root_2x ? 2 : 1;
160  int pll_seld52x = pll_seld52x_map[pll_seld5];
161 
162  int VCO = (xclk / 1000) * pll_multiplier * pll_root_div * 2 / pll_pre_div2x;
163  int PLLCLK = pll_bypass ? (xclk) : (VCO * 1000 * 2 / pll_sys_div / pll_seld52x);
164  int PCLK = PLLCLK / 2 / ((pclk_manual && pclk_div) ? pclk_div : 1);
165  int SYSCLK = PLLCLK / 4;
166 
167  ESP_LOGI(TAG, "Calculated VCO: %d Hz, PLLCLK: %d Hz, SYSCLK: %d Hz, PCLK: %d Hz", VCO * 1000,
168  PLLCLK, SYSCLK, PCLK);
169  return SYSCLK;
170 }

172 static int set_pll(sensor_t* sensor, bool bypass, uint8_t multiplier, uint8_t sys_div,
173  uint8_t pre_div, bool root_2x, uint8_t seld5, bool pclk_manual, uint8_t pclk_div)
174 {
175  int ret = 0;
176  if(multiplier > 31 || sys_div > 15 || pre_div > 3 || pclk_div > 31 || seld5 > 3)
177  {
178  ESP_LOGE(TAG, "Invalid arguments");
179  return -1;
180  }
181 
182  calc_sysclk(sensor->xclk_freq_hz, bypass, multiplier, sys_div, pre_div, root_2x, seld5,
183  pclk_manual, pclk_div);
184 
185  ret = write_reg(sensor->slv_addr, SC_PLLS_CTRL0, bypass ? 0x80 : 0x00);
186  if(ret == 0)
187  {
188  ret = write_reg(sensor->slv_addr, SC_PLLS_CTRL1, multiplier & 0x1f);
189  }
190  if(ret == 0)
191  {
192  ret = write_reg(sensor->slv_addr, SC_PLLS_CTRL2, 0x10 | (sys_div & 0x0f));
193  }
194  if(ret == 0)
195  {
196  ret = write_reg(sensor->slv_addr, SC_PLLS_CTRL3,
197  (pre_div & 0x3) << 4 | seld5 | (root_2x ? 0x40 : 0x00));
198  }
199  if(ret == 0)
200  {
201  ret = write_reg(sensor->slv_addr, PCLK_RATIO, pclk_div & 0x1f);
202  }
203  if(ret == 0)
204  {
205  ret = write_reg(sensor->slv_addr, VFIFO_CTRL0C, pclk_manual ? 0x22 : 0x20);
206  }
207  if(ret)
208  {
209  ESP_LOGE(TAG, "set_sensor_pll FAILED!");
210  }
211  return ret;
212 }

279 static int set_image_options(sensor_t* sensor)
280 {
281  int ret = 0;
282  uint8_t reg20 = 0;
283  uint8_t reg21 = 0;
284  uint8_t reg4514 = 0;
285  uint8_t reg4514_test = 0;
286 
287  // compression
288  if(sensor->pixformat == PIXFORMAT_JPEG)
289  {
290  reg21 |= 0x20;
291  }
292 
293  // binning
294  if(sensor->status.binning)
295  {
296  reg20 |= 0x01;
297  reg21 |= 0x01;
298  reg4514_test |= 4;
299  }
300  else
301  {
302  reg20 |= 0x40;
303  }
304 
305  // V-Flip
306  if(sensor->status.vflip)
307  {
308  reg20 |= 0x06;
309  reg4514_test |= 1;
310  }
311 
312  // H-Mirror
313  if(sensor->status.hmirror)
314  {
315  reg21 |= 0x06;
316  reg4514_test |= 2;
317  }
318 
319  switch(reg4514_test)
320  {
321  // no binning
322  case 0:
323  reg4514 = 0x88;
324  break; // normal
325  case 1:
326  reg4514 = 0x88;
327  break; // v-flip
328  case 2:
329  reg4514 = 0xbb;
330  break; // h-mirror
331  case 3:
332  reg4514 = 0xbb;
333  break; // v-flip+h-mirror
334  // binning
335  case 4:
336  reg4514 = 0xaa;
337  break; // normal
338  case 5:
339  reg4514 = 0xbb;
340  break; // v-flip
341  case 6:
342  reg4514 = 0xbb;
343  break; // h-mirror
344  case 7:
345  reg4514 = 0xaa;
346  break; // v-flip+h-mirror
347  }
348 
349  if(write_reg(sensor->slv_addr, TIMING_TC_REG20, reg20) ||
350  write_reg(sensor->slv_addr, TIMING_TC_REG21, reg21) ||
351  write_reg(sensor->slv_addr, 0x4514, reg4514))
352  {
353  ESP_LOGE(TAG, "Setting Image Options Failed");
354  ret = -1;
355  }
356 
357  if(sensor->status.binning)
358  {
359  ret = write_reg(sensor->slv_addr, 0x4520, 0x0b) ||
360  write_reg(sensor->slv_addr, X_INCREMENT, 0x31) // odd:3, even: 1
361  || write_reg(sensor->slv_addr, Y_INCREMENT, 0x31); // odd:3, even: 1
362  }
363  else
364  {
365  ret = write_reg(sensor->slv_addr, 0x4520, 0xb0) ||
366  write_reg(sensor->slv_addr, X_INCREMENT, 0x11) // odd:1, even: 1
367  || write_reg(sensor->slv_addr, Y_INCREMENT, 0x11); // odd:1, even: 1
368  }
369 
370  ESP_LOGD(TAG,
371  "Set Image Options: Compression: %u, Binning: %u, V-Flip: %u, H-Mirror: %u, Reg-4514: "
372  "0x%02x",
373  sensor->pixformat == PIXFORMAT_JPEG, sensor->status.binning, sensor->status.vflip,
374  sensor->status.hmirror, reg4514);
375  return ret;
376 }

378 static int set_framesize(sensor_t* sensor, framesize_t framesize)
379 {
380  int ret = 0;
381 
382  if(framesize > FRAMESIZE_QXGA)
383  {
384  ESP_LOGW(TAG, "Invalid framesize: %u", framesize);
385  framesize = FRAMESIZE_QXGA;
386  }
387  framesize_t old_framesize = sensor->status.framesize;
388  sensor->status.framesize = framesize;
389  uint16_t w = resolution[framesize].width;
390  uint16_t h = resolution[framesize].height;
391  aspect_ratio_t ratio = resolution[sensor->status.framesize].aspect_ratio;
392  ratio_settings_t settings = ratio_table[ratio];
393 
394  sensor->status.binning = (w <= (settings.max_width / 2) && h <= (settings.max_height / 2));
395  sensor->status.scale = !((w == settings.max_width && h == settings.max_height) ||
396  (w == (settings.max_width / 2) && h == (settings.max_height / 2)));
397 
398  ret = write_addr_reg(sensor->slv_addr, X_ADDR_ST_H, settings.start_x, settings.start_y) ||
399  write_addr_reg(sensor->slv_addr, X_ADDR_END_H, settings.end_x, settings.end_y) ||
400  write_addr_reg(sensor->slv_addr, X_OUTPUT_SIZE_H, w, h);
401 
402  if(ret)
403  {
404  goto fail;
405  }
406 
407  if(sensor->status.binning)
408  {
409  ret = write_addr_reg(sensor->slv_addr, X_TOTAL_SIZE_H, settings.total_x,
410  (settings.total_y / 2) + 1) ||
411  write_addr_reg(sensor->slv_addr, X_OFFSET_H, 8, 2);
412  }
413  else
414  {
415  ret =
416  write_addr_reg(sensor->slv_addr, X_TOTAL_SIZE_H, settings.total_x, settings.total_y) ||
417  write_addr_reg(sensor->slv_addr, X_OFFSET_H, 16, 6);
418  }
419 
420  if(ret == 0)
421  {
422  ret = write_reg_bits(sensor->slv_addr, ISP_CONTROL_01, 0x20, sensor->status.scale);
423  }
424 
425  if(ret == 0)
426  {
427  ret = set_image_options(sensor);
428  }
429 
430  if(ret)
431  {
432  goto fail;
433  }
434 
435  if(sensor->pixformat == PIXFORMAT_JPEG)
436  {
437  if(framesize == FRAMESIZE_QXGA || sensor->xclk_freq_hz == 16000000)
438  {
439  // 40MHz SYSCLK and 10MHz PCLK
440  ret = set_pll(sensor, false, 24, 1, 3, false, 0, true, 8);
441  }
442  else
443  {
444  // 50MHz SYSCLK and 10MHz PCLK
445  ret = set_pll(sensor, false, 30, 1, 3, false, 0, true, 10);
446  }
447  }
448  else
449  {
450  // tuned for 16MHz XCLK and 8MHz PCLK
451  if(framesize > FRAMESIZE_HVGA)
452  {
453  // 8MHz SYSCLK and 8MHz PCLK (4.44 FPS)
454  ret = set_pll(sensor, false, 4, 1, 0, false, 2, true, 2);
455  }
456  else if(framesize >= FRAMESIZE_QVGA)
457  {
458  // 16MHz SYSCLK and 8MHz PCLK (10.25 FPS)
459  ret = set_pll(sensor, false, 8, 1, 0, false, 2, true, 4);
460  }
461  else
462  {
463  // 32MHz SYSCLK and 8MHz PCLK (17.77 FPS)
464  ret = set_pll(sensor, false, 8, 1, 0, false, 0, true, 8);
465  }
466  }
467 
468  if(ret == 0)
469  {
470  ESP_LOGD(TAG, "Set framesize to: %ux%u", w, h);
471  }
472  return ret;
473 
474 fail:
475  sensor->status.framesize = old_framesize;
476  ESP_LOGE(TAG, "Setting framesize to: %ux%u failed", w, h);
477  return ret;
478 }

639 static int get_agc_gain(sensor_t* sensor)
640 {
641  int ra = read_reg(sensor->slv_addr, 0x350a);
642  if(ra < 0)
643  {
644  return 0;
645  }
646  int rb = read_reg(sensor->slv_addr, 0x350b);
647  if(rb < 0)
648  {
649  return 0;
650  }
651  int res = (rb & 0xF0) >> 4 | (ra & 0x03) << 4;
652  if(rb & 0x0F)
653  {
654  res += 1;
655  }
656  return res;
657 }

659 // real gain
660 static int set_agc_gain(sensor_t* sensor, int gain)
661 {
662  int ret = 0;
663  if(gain < 0)
664  {
665  gain = 0;
666  }
667  else if(gain > 64)
668  {
669  gain = 64;
670  }
671 
672  // gain value is 6.4 bits float
673  // in order to use the max range, we deduct 1/16
674  int gainv = gain << 4;
675  if(gainv)
676  {
677  gainv -= 1;
678  }
