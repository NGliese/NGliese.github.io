248 /* STATIC */ esp_err_t WiFi::eventHandler(void* ctx, system_event_t* event)
249 {
250  // This is the common event handler that we have provided for all event processing. It is
251  // called for every event that is received by the WiFi subsystem. The "ctx" parameter is an
252  // instance of the current WiFi object that we are processing. We can then retrieve the
253  // specific/custom event handler from within it and invoke that. This then makes this an
254  // indirection vector to the real caller.
255 
256  WiFi* pWiFi = (WiFi*)ctx; // retrieve the WiFi object from the passed in context.
257 
258  // Invoke the event handler.
259  esp_err_t rc;
260  if(pWiFi->m_pWifiEventHandler != nullptr)
261  {
262  rc = pWiFi->m_pWifiEventHandler->getEventHandler()(pWiFi->m_pWifiEventHandler, event);
263  }
264  else
265  {
266  rc = ESP_OK;
267  }
268 
269  // If the event we received indicates that we now have an IP address or that a connection was
270  // disconnected then unlock the mutex that indicates we are waiting for a connection complete.
271  if(event->event_id == SYSTEM_EVENT_STA_GOT_IP ||
272  event->event_id == SYSTEM_EVENT_STA_DISCONNECTED)
273  {
274  if(event->event_id == SYSTEM_EVENT_STA_GOT_IP)
275  { // If we connected and have an IP, change the status to ESP_OK. Otherwise, change it to
276  // the reason code.
277  pWiFi->m_apConnectionStatus = ESP_OK;
278  }
279  else
280  {
281  pWiFi->m_apConnectionStatus = event->event_info.disconnected.reason;
282  }
283  pWiFi->m_connectFinished.give();
284  }
285 
286  return rc;
287 } // eventHandler
