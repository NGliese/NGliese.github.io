86 static esp_err_t camera_probe(const camera_config_t* config, camera_model_t* out_camera_model)
87 {
88  *out_camera_model = CAMERA_NONE;
89  if(s_state != NULL)
90  {
91  return ESP_ERR_INVALID_STATE;
92  }
93 
94  s_state = (camera_state_t*)calloc(sizeof(camera_state_t), 1);
95  if(!s_state)
96  {
97  return ESP_ERR_NO_MEM;
98  }
99 
100  if(config->pin_xclk >= 0)
101  {
102  ESP_LOGD(TAG, "Enabling XCLK output");
103  CAMERA_ENABLE_OUT_CLOCK(config);
104  }
105 
106  if(config->pin_sscb_sda != -1)
107  {
108  ESP_LOGD(TAG, "Initializing SSCB");
109  SCCB_Init(config->pin_sscb_sda, config->pin_sscb_scl);
110  }
111 
112  if(config->pin_pwdn >= 0)
113  {
114  ESP_LOGD(TAG, "Resetting camera by power down line");
115  gpio_config_t conf = {0};
116  conf.pin_bit_mask = 1LL << config->pin_pwdn;
117  conf.mode = GPIO_MODE_OUTPUT;
118  gpio_config(&conf);
119 
120  // carefull, logic is inverted compared to reset pin
121  gpio_set_level(config->pin_pwdn, 1);
122  vTaskDelay(10 / portTICK_PERIOD_MS);
123  gpio_set_level(config->pin_pwdn, 0);
124  vTaskDelay(10 / portTICK_PERIOD_MS);
125  }
126 
127  if(config->pin_reset >= 0)
128  {
129  ESP_LOGD(TAG, "Resetting camera");
130  gpio_config_t conf = {0};
131  conf.pin_bit_mask = 1LL << config->pin_reset;
132  conf.mode = GPIO_MODE_OUTPUT;
133  gpio_config(&conf);
134 
135  gpio_set_level(config->pin_reset, 0);
136  vTaskDelay(10 / portTICK_PERIOD_MS);
137  gpio_set_level(config->pin_reset, 1);
138  vTaskDelay(10 / portTICK_PERIOD_MS);
139  }
140 
141  ESP_LOGD(TAG, "Searching for camera address");
142  vTaskDelay(10 / portTICK_PERIOD_MS);
143 
144  uint8_t slv_addr = SCCB_Probe();
145 
146  if(slv_addr == 0)
147  {
148  CAMERA_DISABLE_OUT_CLOCK();
149  return ESP_ERR_NOT_FOUND;
150  }
151 
152  ESP_LOGI(TAG, "Detected camera at address=0x%02x", slv_addr);
153  s_state->sensor.slv_addr = slv_addr;
154  s_state->sensor.xclk_freq_hz = config->xclk_freq_hz;
155 
161  sensor_id_t* id = &s_state->sensor.id;
162  for(size_t i = 0; i < sizeof(g_sensors) / sizeof(sensor_func_t); i++)
163  {
164  if(g_sensors[i].detect(slv_addr, id))
165  {
166  camera_sensor_info_t* info = esp_camera_sensor_get_info(id);
167  if(NULL != info)
168  {
169  *out_camera_model = info->model;
170  ESP_LOGI(TAG, "Detected %s camera", info->name);
171  g_sensors[i].init(&s_state->sensor);
172  break;
173  }
174  }
175  }
176 
177  if(CAMERA_NONE == *out_camera_model)
178  { // If no supported sensors are detected
179  CAMERA_DISABLE_OUT_CLOCK();
180  ESP_LOGE(TAG, "Detected camera not supported.");
181  return ESP_ERR_NOT_SUPPORTED;
182  }
183 
184  ESP_LOGI(TAG, "Camera PID=0x%02x VER=0x%02x MIDL=0x%02x MIDH=0x%02x", id->PID, id->VER,
185  id->MIDH, id->MIDL);
186 
187  ESP_LOGD(TAG, "Doing SW reset of sensor");
188  vTaskDelay(10 / portTICK_PERIOD_MS);
189  s_state->sensor.reset(&s_state->sensor);
190 
191  return ESP_OK;
192 }

194 esp_err_t esp_camera_init(const camera_config_t* config)
195 {
196  esp_err_t err;
197  err = cam_init(config);
198  if(err != ESP_OK)
199  {
200  ESP_LOGE(TAG, "Camera init failed with error 0x%x", err);
201  return err;
202  }
203 
204  camera_model_t camera_model = CAMERA_NONE;
205  err = camera_probe(config, &camera_model);
206  if(err != ESP_OK)
207  {
208  ESP_LOGE(TAG, "Camera probe failed with error 0x%x(%s)", err, esp_err_to_name(err));
209  goto fail;
210  }
211 
212  framesize_t frame_size = (framesize_t)config->frame_size;
213  pixformat_t pix_format = (pixformat_t)config->pixel_format;
214 
215  if(PIXFORMAT_JPEG == pix_format && (!camera_sensor[camera_model].support_jpeg))
216  {
217  ESP_LOGE(TAG, "JPEG format is not supported on this sensor");
218  err = ESP_ERR_NOT_SUPPORTED;
219  goto fail;
220  }
221 
222  if(frame_size > camera_sensor[camera_model].max_size)
223  {
224  ESP_LOGW(TAG, "The frame size exceeds the maximum for this sensor, it will be forced to "
225  "the maximum possible value");
226  frame_size = camera_sensor[camera_model].max_size;
227  }
228 
229  err = cam_config(config, frame_size, s_state->sensor.id.PID);
230  if(err != ESP_OK)
231  {
232  ESP_LOGE(TAG, "Camera config failed with error 0x%x", err);
233  goto fail;
234  }
235 
236  s_state->sensor.status.framesize = frame_size;
237  s_state->sensor.pixformat = pix_format;
238  ESP_LOGD(TAG, "Setting frame size to %dx%d", resolution[frame_size].width,
239  resolution[frame_size].height);
240  if(s_state->sensor.set_framesize(&s_state->sensor, frame_size) != 0)
241  {
242  ESP_LOGE(TAG, "Failed to set frame size");
243  err = ESP_ERR_CAMERA_FAILED_TO_SET_FRAME_SIZE;
244  goto fail;
245  }
246  s_state->sensor.set_pixformat(&s_state->sensor, pix_format);
247 
248  if(s_state->sensor.id.PID == OV2640_PID)
249  {
250  s_state->sensor.set_gainceiling(&s_state->sensor, GAINCEILING_2X);
251  s_state->sensor.set_bpc(&s_state->sensor, false);
252  s_state->sensor.set_wpc(&s_state->sensor, true);
253  s_state->sensor.set_lenc(&s_state->sensor, true);
254  }
255 
256  if(pix_format == PIXFORMAT_JPEG)
257  {
258  s_state->sensor.set_quality(&s_state->sensor, config->jpeg_quality);
259  }
260  s_state->sensor.init_status(&s_state->sensor);
261 
262  cam_start();
263 
264  return ESP_OK;
265 
266 fail:
267  esp_camera_deinit();
268  return err;
269 }

272 {
273  esp_err_t ret = cam_deinit();

323 esp_err_t esp_camera_save_to_nvs(const char* key)
324 {
325 #if ESP_IDF_VERSION_MAJOR > 3
326  nvs_handle_t handle;
327 #else
328  nvs_handle handle;
329 #endif
330  esp_err_t ret = nvs_open(key, NVS_READWRITE, &handle);
331 
332  if(ret == ESP_OK)
333  {
334  sensor_t* s = esp_camera_sensor_get();
335  if(s != NULL)
336  {
337  ret = nvs_set_blob(handle, CAMERA_SENSOR_NVS_KEY, &s->status, sizeof(camera_status_t));
338  if(ret == ESP_OK)
339  {
340  uint8_t pf = s->pixformat;
341  ret = nvs_set_u8(handle, CAMERA_PIXFORMAT_NVS_KEY, pf);
342  }
343  return ret;
344  }
345  else
346  {
347  return ESP_ERR_CAMERA_NOT_DETECTED;
348  }
349  nvs_close(handle);
350  return ret;
351  }
352  else
353  {
354  return ret;
355  }
356 }
