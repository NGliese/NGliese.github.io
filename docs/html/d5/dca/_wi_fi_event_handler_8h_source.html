7  * A WiFiEventHandler defines a class that has methods that will be called back when a WiFi event is
8  * detected.
9  *
10  * Typically this class is subclassed to provide implementations for the callbacks we want to
11  * handle.
12  *
13  * class MyHandler: public WiFiEventHandler {
14  * esp_err_t apStart() {
15  * ESP_LOGD(tag, "MyHandler(Class): apStart");
16  * return ESP_OK;
17  * }
18  * }
19  *
20  * WiFi wifi;
21  * MyHandler *eventHandler = new MyHandler();
22  * wifi.setWifiEventHandler(eventHandler);
23  * wifi.startAP("MYSSID", "password");
24  *
25  * The overridable functions are:
26  * * esp_err_t apStaConnected(system_event_ap_staconnected_t info)
27  * * esp_err_t apStaDisconnected(system_event_ap_stadisconnected_t info)
28  * * esp_err_t apStart()
29  * * esp_err_t apStop()
30  * * esp_err_t staConnected(system_event_sta_connected_t info)
31  * * esp_err_t staDisconnected(system_event_sta_disconnected_t info)
32  * * esp_err_t staGotIp(system_event_sta_got_ip_t info)
33  * * esp_err_t staScanDone(system_event_sta_scan_done_t info)
34  * * esp_err_t staAuthChange(system_event_sta_authmode_change_t info)
35  * * esp_err_t staStart()
36  * * esp_err_t staStop()
37  * * esp_err_t wifiReady()
38  */
