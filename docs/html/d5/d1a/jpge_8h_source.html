31  inline bool check() const

41  return true;
42  }

44  // Quality: 1-100, higher is better. Typical values are around 50-95.
45  int m_quality;
46 
47  // m_subsampling:
48  // 0 = Y (grayscale) only
49  // 1 = H1V1 subsampling (YCbCr 1x1x1, 3 blocks per MCU)
50  // 2 = H2V1 subsampling (YCbCr 2x1x1, 4 blocks per MCU)
51  // 3 = H2V2 subsampling (YCbCr 4x1x1, 6 blocks per MCU-- very common)
52  subsampling_t m_subsampling;
53 };
