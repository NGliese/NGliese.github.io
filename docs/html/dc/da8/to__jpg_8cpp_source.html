42 static IRAM_ATTR void convert_line_format(uint8_t* src, pixformat_t format, uint8_t* dst,
43  size_t width, size_t in_channels, size_t line)
44 {
45  int i = 0, o = 0, l = 0;
46  if(format == PIXFORMAT_GRAYSCALE)
47  {
48  memcpy(dst, src + line * width, width);
49  }
50  else if(format == PIXFORMAT_RGB888)
51  {
52  l = width * 3;
53  src += l * line;
54  for(i = 0; i < l; i += 3)
55  {
56  dst[o++] = src[i + 2];
57  dst[o++] = src[i + 1];
58  dst[o++] = src[i];
59  }
60  }
61  else if(format == PIXFORMAT_RGB565)
62  {
63  l = width * 2;
64  src += l * line;
65  for(i = 0; i < l; i += 2)
66  {
67  dst[o++] = src[i] & 0xF8;
68  dst[o++] = (src[i] & 0x07) << 5 | (src[i + 1] & 0xE0) >> 3;
69  dst[o++] = (src[i + 1] & 0x1F) << 3;
70  }
71  }
72  else if(format == PIXFORMAT_YUV422)
73  {
74  uint8_t y0, y1, u, v;
75  uint8_t r, g, b;
76  l = width * 2;
77  src += l * line;
78  for(i = 0; i < l; i += 4)
79  {
80  y0 = src[i];
81  u = src[i + 1];
82  y1 = src[i + 2];
83  v = src[i + 3];
84 
85  yuv2rgb(y0, u, v, &r, &g, &b);
86  dst[o++] = r;
87  dst[o++] = g;
88  dst[o++] = b;
89 
90  yuv2rgb(y1, u, v, &r, &g, &b);
91  dst[o++] = r;
92  dst[o++] = g;
93  dst[o++] = b;
94  }
95  }
96 }

98 bool convert_image(uint8_t* src, uint16_t width, uint16_t height, pixformat_t format,
99  uint8_t quality, jpge::output_stream* dst_stream)
100 {
101  int num_channels = 3;
102  jpge::subsampling_t subsampling = jpge::H2V2;
103 
104  if(format == PIXFORMAT_GRAYSCALE)
105  {
106  num_channels = 1;
107  subsampling = jpge::Y_ONLY;
108  }
109 
110  if(!quality)
111  {
112  quality = 1;
113  }
114  else if(quality > 100)
115  {
116  quality = 100;
117  }
118 
119  jpge::params comp_params = jpge::params();
120  comp_params.m_subsampling = subsampling;
121  comp_params.m_quality = quality;
122 
123  jpge::jpeg_encoder dst_image;
124 
125  if(!dst_image.init(dst_stream, width, height, num_channels, comp_params))
126  {
127  ESP_LOGE(TAG, "JPG encoder init failed");
128  return false;
129  }
130 
131  uint8_t* line = (uint8_t*)_malloc(width * num_channels);
132  if(!line)
133  {
134  ESP_LOGE(TAG, "Scan line malloc failed");
135  return false;
136  }
137 
138  for(int i = 0; i < height; i++)
139  {
140  convert_line_format(src, format, line, width, num_channels, i);
141  if(!dst_image.process_scanline(line))
142  {
143  ESP_LOGE(TAG, "JPG process line %u failed", i);
144  free(line);
145  return false;
146  }
147  }
148  free(line);
149 
150  if(!dst_image.process_scanline(NULL))
151  {
152  ESP_LOGE(TAG, "JPG image finish failed");
153  return false;
154  }
155  dst_image.deinit();
156  return true;
157 }
