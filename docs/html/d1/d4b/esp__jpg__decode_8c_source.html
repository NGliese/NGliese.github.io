44 static uint32_t _jpg_write(JDEC* decoder, void* bitmap, JRECT* rect)
45 {
46  uint16_t x = rect->left;
47  uint16_t y = rect->top;
48  uint16_t w = rect->right + 1 - x;
49  uint16_t h = rect->bottom + 1 - y;
50  uint8_t* data = (uint8_t*)bitmap;
51 
52  esp_jpg_decoder_t* jpeg = (esp_jpg_decoder_t*)decoder->device;
53 
54  if(jpeg->writer)
55  {
56  return jpeg->writer(jpeg->arg, x, y, w, h, data);
57  }
58  return 0;
59 }

61 static uint32_t _jpg_read(JDEC* decoder, uint8_t* buf, uint32_t len)
62 {
63  esp_jpg_decoder_t* jpeg = (esp_jpg_decoder_t*)decoder->device;
64  if(jpeg->len && len > (jpeg->len - jpeg->index))
65  {
66  len = jpeg->len - jpeg->index;
67  }
68  if(len)
69  {
70  len = jpeg->reader(jpeg->arg, jpeg->index, buf, len);
71  if(!len)
72  {
73  ESP_LOGE(TAG, "Read Fail at %u/%u", jpeg->index, jpeg->len);
74  }
75  jpeg->index += len;
76  }
77  return len;
78 }

80 esp_err_t esp_jpg_decode(size_t len, jpg_scale_t scale, jpg_reader_cb reader, jpg_writer_cb writer,
81  void* arg)
82 {
83  static uint8_t work[3100];
84  JDEC decoder;
85  esp_jpg_decoder_t jpeg;
86 
87  jpeg.len = len;
88  jpeg.reader = reader;
89  jpeg.writer = writer;
90  jpeg.arg = arg;
91  jpeg.scale = scale;
92  jpeg.index = 0;
93 
94  JRESULT jres = jd_prepare(&decoder, _jpg_read, work, 3100, &jpeg);
95  if(jres != JDR_OK)
96  {
97  ESP_LOGE(TAG, "JPG Header Parse Failed! %s", jd_errors[jres]);
98  return ESP_FAIL;
99  }
100 
101  uint16_t output_width = decoder.width / (1 << (uint8_t)(jpeg.scale));
102  uint16_t output_height = decoder.height / (1 << (uint8_t)(jpeg.scale));
103 
104  // output start
105  writer(arg, 0, 0, output_width, output_height, NULL);
106  // output write
107  jres = jd_decomp(&decoder, _jpg_write, (uint8_t)jpeg.scale);
108  // output end
109  writer(arg, output_width, output_height, output_width, output_height, NULL);
110 
111  if(jres != JDR_OK)
112  {
113  ESP_LOGE(TAG, "JPG Decompression Failed! %s", jd_errors[jres]);
114  return ESP_FAIL;
115  }
116  // check if all data has been consumed.
117  if(len && jpeg.index < len)
118  {
119  _jpg_read(&decoder, NULL, len - jpeg.index);
120  }
121 
122  return ESP_OK;
123 }
