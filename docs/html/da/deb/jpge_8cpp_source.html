206 static void DCT2D(int32* p)
207 {
208  int32 c, *q = p;
209  for(c = 7; c >= 0; c--, q += 8)
210  {
211  int32 s0 = q[0], s1 = q[1], s2 = q[2], s3 = q[3], s4 = q[4], s5 = q[5], s6 = q[6],
212  s7 = q[7];
213  DCT1D(s0, s1, s2, s3, s4, s5, s6, s7);
214  q[0] = s0 << ROW_BITS;
215  q[1] = DCT_DESCALE(s1, CONST_BITS - ROW_BITS);
216  q[2] = DCT_DESCALE(s2, CONST_BITS - ROW_BITS);
217  q[3] = DCT_DESCALE(s3, CONST_BITS - ROW_BITS);
218  q[4] = s4 << ROW_BITS;
219  q[5] = DCT_DESCALE(s5, CONST_BITS - ROW_BITS);
220  q[6] = DCT_DESCALE(s6, CONST_BITS - ROW_BITS);
221  q[7] = DCT_DESCALE(s7, CONST_BITS - ROW_BITS);
222  }
223  for(q = p, c = 7; c >= 0; c--, q++)
224  {
225  int32 s0 = q[0 * 8], s1 = q[1 * 8], s2 = q[2 * 8], s3 = q[3 * 8], s4 = q[4 * 8],
226  s5 = q[5 * 8], s6 = q[6 * 8], s7 = q[7 * 8];
227  DCT1D(s0, s1, s2, s3, s4, s5, s6, s7);
228  q[0 * 8] = DCT_DESCALE(s0, ROW_BITS + 3);
229  q[1 * 8] = DCT_DESCALE(s1, CONST_BITS + ROW_BITS + 3);
230  q[2 * 8] = DCT_DESCALE(s2, CONST_BITS + ROW_BITS + 3);
231  q[3 * 8] = DCT_DESCALE(s3, CONST_BITS + ROW_BITS + 3);
232  q[4 * 8] = DCT_DESCALE(s4, ROW_BITS + 3);

234  q[6 * 8] = DCT_DESCALE(s6, CONST_BITS + ROW_BITS + 3);
235  q[7 * 8] = DCT_DESCALE(s7, CONST_BITS + ROW_BITS + 3);

239 // Compute the actual canonical Huffman codes/code sizes given the JPEG huff bits and val arrays.
240 static void compute_huffman_table(uint* codes, uint8* code_sizes, uint8* bits, uint8* val)
241 {
242  int i, l, last_p, si;
243  static uint8 huff_size[257];
244  static uint huff_code[257];
245  uint code;
246 
247  int p = 0;
248  for(l = 1; l <= 16; l++)
249  {
250  for(i = 1; i <= bits[l]; i++)
251  {
252  huff_size[p++] = (char)l;
253  }
254  }
255 
256  huff_size[p] = 0;
257  last_p = p; // write sentinel
258 
259  code = 0;
260  si = huff_size[0];
261  p = 0;
262 
263  while(huff_size[p])
264  {
265  while(huff_size[p] == si)
266  {
267  huff_code[p++] = code++;
268  }
269  code <<= 1;
270  si++;
271  }
272 
273  memset(codes, 0, sizeof(codes[0]) * 256);
274  memset(code_sizes, 0, sizeof(code_sizes[0]) * 256);
275  for(p = 0; p < last_p; p++)
276  {
277  codes[val[p]] = huff_code[p];
278  code_sizes[val[p]] = huff_size[p];
279  }
280 }

539 void jpeg_encoder::code_coefficients_pass_two(int component_num)
540 {
541  int i, j, run_len, nbits, temp1, temp2;
542  int16* pSrc = m_coefficient_array;
543  uint* codes[2];
544  uint8* code_sizes[2];
545 
546  if(component_num == 0)
547  {
548  codes[0] = m_huff_codes[0 + 0];
549  codes[1] = m_huff_codes[2 + 0];
550  code_sizes[0] = m_huff_code_sizes[0 + 0];
551  code_sizes[1] = m_huff_code_sizes[2 + 0];
552  }
553  else
554  {
555  codes[0] = m_huff_codes[0 + 1];
556  codes[1] = m_huff_codes[2 + 1];
557  code_sizes[0] = m_huff_code_sizes[0 + 1];
558  code_sizes[1] = m_huff_code_sizes[2 + 1];
559  }
560 
561  temp1 = temp2 = pSrc[0] - m_last_dc_val[component_num];
562  m_last_dc_val[component_num] = pSrc[0];
563 
564  if(temp1 < 0)
565  {
566  temp1 = -temp1;
567  temp2--;
568  }
569 
570  nbits = 0;
571  while(temp1)
572  {
573  nbits++;
574  temp1 >>= 1;
575  }
576 
577  put_bits(codes[0][nbits], code_sizes[0][nbits]);
578  if(nbits)
579  put_bits(temp2 & ((1 << nbits) - 1), nbits);
580 
581  for(run_len = 0, i = 1; i < 64; i++)
582  {
583  if((temp1 = m_coefficient_array[i]) == 0)
584  run_len++;
585  else
586  {
587  while(run_len >= 16)
588  {
589  put_bits(codes[1][0xF0], code_sizes[1][0xF0]);
590  run_len -= 16;
591  }
592  if((temp2 = temp1) < 0)
593  {
594  temp1 = -temp1;
595  temp2--;
596  }
597  nbits = 1;
598  while(temp1 >>= 1)
599  nbits++;
600  j = (run_len << 4) + nbits;
601  put_bits(codes[1][j], code_sizes[1][j]);
602  put_bits(temp2 & ((1 << nbits) - 1), nbits);
603  run_len = 0;
604  }
605  }
606  if(run_len)
607  put_bits(codes[1][0], code_sizes[1][0]);
608 }

673 void jpeg_encoder::load_mcu(const void* pSrc)
674 {
675  const uint8* Psrc = reinterpret_cast<const uint8*>(pSrc);
676 
677  uint8* pDst = m_mcu_lines[m_mcu_y_ofs]; // OK to write up to m_image_bpl_xlt bytes to pDst
678 
679  if(m_num_components == 1)
680  {
681  if(m_image_bpp == 3)
682  RGB_to_Y(pDst, Psrc, m_image_x);
683  else
684  memcpy(pDst, Psrc, m_image_x);
685  }
686  else
687  {
688  if(m_image_bpp == 3)
689  RGB_to_YCC(pDst, Psrc, m_image_x);
690  else
691  Y_to_YCC(pDst, Psrc, m_image_x);
692  }
693 
694  // Possibly duplicate pixels at end of scanline if not a multiple of 8 or 16
695  if(m_num_components == 1)
696  memset(m_mcu_lines[m_mcu_y_ofs] + m_image_bpl_xlt, pDst[m_image_bpl_xlt - 1],
697  m_image_x_mcu - m_image_x);
698  else
699  {
700  const uint8 y = pDst[m_image_bpl_xlt - 3 + 0], cb = pDst[m_image_bpl_xlt - 3 + 1],
701  cr = pDst[m_image_bpl_xlt - 3 + 2];
702  uint8* q = m_mcu_lines[m_mcu_y_ofs] + m_image_bpl_xlt;
703  for(int i = m_image_x; i < m_image_x_mcu; i++)
704  {
705  *q++ = y;
706  *q++ = cb;
707  *q++ = cr;
708  }
709  }
710 
711  if(++m_mcu_y_ofs == m_mcu_y)
712  {
713  process_mcu_row();
714  m_mcu_y_ofs = 0;
715  }

739  {

759  case H2V1: {

762  m_comp_h_samp[1] = 1;
763  m_comp_v_samp[1] = 1;
764  m_comp_h_samp[2] = 1;
765  m_comp_v_samp[2] = 1;

773  m_comp_h_samp[1] = 1;
774  m_comp_v_samp[1] = 1;
775  m_comp_h_samp[2] = 1;
776  m_comp_v_samp[2] = 1;

787  m_image_y_mcu = (m_image_y + m_mcu_y - 1) & (~(m_mcu_y - 1));

803  compute_quant_table(m_quantization_tables[1], s_std_croma_quant);

808  m_huff_initialized = true;
