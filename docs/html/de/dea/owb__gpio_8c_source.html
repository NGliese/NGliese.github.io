73 // 1-Wire timing delays (standard) in microseconds.
74 // Labels and values are from https://www.maximintegrated.com/en/app-notes/index.mvp/id/126
75 static const struct _OneWireBus_Timing _StandardTiming = {
76  6, // A - read/write "1" master pull DQ low duration
77  64, // B - write "0" master pull DQ low duration
78  60, // C - write "1" master pull DQ high duration
79  10, // D - write "0" master pull DQ high duration
80  9, // E - read master pull DQ high duration
81  55, // F - complete read timeslot + 10ms recovery
82  0, // G - wait before reset
83  480, // H - master pull DQ low duration
84  70, // I - master pull DQ high duration
85  410, // J - complete presence timeslot + recovery
86 };

103 static owb_status _reset(const OneWireBus* bus, bool* is_present)
104 {
105  bool present = false;
106  portMUX_TYPE timeCriticalMutex = portMUX_INITIALIZER_UNLOCKED;
107  portENTER_CRITICAL(&timeCriticalMutex);
108 
109  owb_gpio_driver_info* i = info_from_bus(bus);
110 
111  gpio_set_direction(i->gpio, GPIO_MODE_OUTPUT);
112  _us_delay(bus->timing->G);
113  gpio_set_level(i->gpio, 0); // Drive DQ low
114  _us_delay(bus->timing->H);
115  gpio_set_direction(i->gpio, GPIO_MODE_INPUT); // Release the bus
116  gpio_set_level(i->gpio, 1); // Reset the output level for the next output
117  _us_delay(bus->timing->I);
118 
119 #ifdef PHY_DEBUG
120  gpio_set_level(PHY_DEBUG_GPIO, 1);
121 #endif
122 
123  int level1 = gpio_get_level(i->gpio);
124 
125 #ifdef PHY_DEBUG
126  gpio_set_level(PHY_DEBUG_GPIO, 0);
127 #endif
128 
129  _us_delay(bus->timing->J); // Complete the reset sequence recovery
130 
131 #ifdef PHY_DEBUG
132  gpio_set_level(PHY_DEBUG_GPIO, 1);
133 #endif
134 
135  int level2 = gpio_get_level(i->gpio);
136 
137 #ifdef PHY_DEBUG
138  gpio_set_level(PHY_DEBUG_GPIO, 0);
139 #endif
140 
141  portEXIT_CRITICAL(&timeCriticalMutex);
142 
143  present = (level1 == 0) && (level2 == 1); // Sample for presence pulse from slave
144  ESP_LOGD(TAG, "reset: level1 0x%x, level2 0x%x, present %d", level1, level2, present);
145 
146  *is_present = present;
147 
148  return OWB_STATUS_OK;
149 }

156 static void _write_bit(const OneWireBus* bus, int bit)
157 {
158  int delay1 = bit ? bus->timing->A : bus->timing->C;
159  int delay2 = bit ? bus->timing->B : bus->timing->D;
160  owb_gpio_driver_info* i = info_from_bus(bus);
161 
162  portMUX_TYPE timeCriticalMutex = portMUX_INITIALIZER_UNLOCKED;
163  portENTER_CRITICAL(&timeCriticalMutex);
164 
165  gpio_set_direction(i->gpio, GPIO_MODE_OUTPUT);
166  gpio_set_level(i->gpio, 0); // Drive DQ low
167  _us_delay(delay1);
168  gpio_set_level(i->gpio, 1); // Release the bus
169  _us_delay(delay2);
170 
171  portEXIT_CRITICAL(&timeCriticalMutex);
172 }

178 static int _read_bit(const OneWireBus* bus)
179 {
180  int result = 0;
181  owb_gpio_driver_info* i = info_from_bus(bus);
182 
183  portMUX_TYPE timeCriticalMutex = portMUX_INITIALIZER_UNLOCKED;
184  portENTER_CRITICAL(&timeCriticalMutex);
185 
186  gpio_set_direction(i->gpio, GPIO_MODE_OUTPUT);
187  gpio_set_level(i->gpio, 0); // Drive DQ low
188  _us_delay(bus->timing->A);
189  gpio_set_direction(i->gpio, GPIO_MODE_INPUT); // Release the bus
190  gpio_set_level(i->gpio, 1); // Reset the output level for the next output
191  _us_delay(bus->timing->E);
192 
193 #ifdef PHY_DEBUG
194  gpio_set_level(PHY_DEBUG_GPIO, 1);
195 #endif
196 
197  int level = gpio_get_level(i->gpio);
198 
199 #ifdef PHY_DEBUG
200  gpio_set_level(PHY_DEBUG_GPIO, 0);
201 #endif
202 
203  _us_delay(bus->timing->F); // Complete the timeslot and 10us recovery
204 
205  portEXIT_CRITICAL(&timeCriticalMutex);
206 
207  result = level & 0x01;
208 
209  return result;
210 }

219 static owb_status _write_bits(const OneWireBus* bus, uint8_t data, int number_of_bits_to_write)
220 {
221  ESP_LOGD(TAG, "write 0x%02x", data);
222  for(int i = 0; i < number_of_bits_to_write; ++i)
223  {
224  _write_bit(bus, data & 0x01);
225  data >>= 1;
226  }
227 
228  return OWB_STATUS_OK;
229 }

238 static owb_status _read_bits(const OneWireBus* bus, uint8_t* out, int number_of_bits_to_read)
239 {
240  uint8_t result = 0;
241  for(int i = 0; i < number_of_bits_to_read; ++i)
242  {
243  result >>= 1;
244  if(_read_bit(bus))
245  {
246  result |= 0x80;
247  }
248  }
249  ESP_LOGD(TAG, "read 0x%02x", result);
250  *out = result;
251 
252  return OWB_STATUS_OK;
253 }

267 OneWireBus* owb_gpio_initialize(owb_gpio_driver_info* driver_info, int gpio)
268 {
269  ESP_LOGD(TAG, "%s(): gpio %d\n", __func__, gpio);
270 
271  driver_info->gpio = gpio;
272  driver_info->bus.driver = &gpio_function_table;
273  driver_info->bus.timing = &_StandardTiming;
274  driver_info->bus.strong_pullup_gpio = GPIO_NUM_NC;
275 
276  // platform specific:
277  gpio_pad_select_gpio(driver_info->gpio);
278 
279 #ifdef PHY_DEBUG
280  gpio_config_t io_conf;
281  io_conf.intr_type = GPIO_INTR_DISABLE;
282  io_conf.mode = GPIO_MODE_OUTPUT;
283  io_conf.pin_bit_mask = PHY_DEBUG_GPIO_MASK;
284  io_conf.pull_down_en = GPIO_PULLDOWN_ENABLE;
285  io_conf.pull_up_en = GPIO_PULLUP_DISABLE;
286  ESP_ERROR_CHECK(gpio_config(&io_conf));
287 #endif
288 
289  return &(driver_info->bus);
290 }
